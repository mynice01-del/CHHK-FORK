// ==UserScript==
// @name         치지직 통합 툴: 미리보기 & 오디오 컴프레셔 슬라이더 최종판
// @namespace    http://tampermonkey.net/
// @version      5.1
// @description  라이브 미리보기, 고성능 오디오 컴프레셔 + 실시간 게인 조절 슬라이더 및 30초 자동 새로고침 (저음 완화 반영)
// @author       You
// @match        https://chzzk.naver.com/*
// @require      https://cdn.jsdelivr.net/npm/hls.js@latest
// @grant        none
// ==/UserScript==

/* global Hls */

(function() {
    'use strict';

    // 오디오 엔진 변수들
    let audioCtx = null;
    let compressorNode = null;
    let gainNode = null;
    const mediaSourceMap = new WeakMap();
    const connectedSources = new WeakSet();
    let isCompressorOn = false;
    let currentGainValue = 1.5; // 컴프레셔 ON시 기본 게인 보정값 (1.0 ~ 3.0)

    // 미리보기 엔진 변수들
    let hls = null, hoverTimeout = null, abortController = null;
    let currentChannel = null, currentPreviewUrl = null;
    const liveCache = new Map();


    // --- 오디오 로직 ---
    function getAudioElements() {
        return document.querySelector('.pzp-pc-html-video-player__video') || document.querySelector('video');
    }

    function initAudioEngine(videoElem) {
        if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        // 전문 컴프레서 노드 생성 및 초기화
        if (!compressorNode) {
            compressorNode = audioCtx.createDynamicsCompressor();
            compressorNode.threshold.setValueAtTime(-16, audioCtx.currentTime);
            compressorNode.knee.setValueAtTime(30, audioCtx.currentTime);
            compressorNode.ratio.setValueAtTime(6, audioCtx.currentTime);
            compressorNode.attack.setValueAtTime(0.00, audioCtx.currentTime);
            compressorNode.release.setValueAtTime(0.1, audioCtx.currentTime);
        }

        // 압축 후 줄어든 볼륨을 보정하기 위한 게인 노드
        if (!gainNode) {
            gainNode = audioCtx.createGain();
            gainNode.gain.setValueAtTime(1.0, audioCtx.currentTime);
        }

        // 노드 연결 구조: 오디오 소스 -> 컴프레서 -> 게인(증폭) -> 스피커 출력
        if (!mediaSourceMap.has(videoElem)) {
            mediaSourceMap.set(videoElem, audioCtx.createMediaElementSource(videoElem));
        }

        const source = mediaSourceMap.get(videoElem);
        if (!connectedSources.has(source)) {
            source.connect(compressorNode);
            compressorNode.connect(gainNode);
            gainNode.connect(audioCtx.destination);
            connectedSources.add(source);
        }
    }

    function toggleCompressor(btn) {
        const video = getAudioElements();
        if (!video) return;
        initAudioEngine(video);

        if (audioCtx.state === 'suspended') audioCtx.resume();

        isCompressorOn = !isCompressorOn;
        const slider = document.getElementById('chzzk-compressor-gain-slider');

        if (isCompressorOn) {
            // 컴프레서 켬: 모든 설정값을 완벽하게 동기화하여 실시간 적용
            compressorNode.threshold.setValueAtTime(-16, audioCtx.currentTime);
            compressorNode.knee.setValueAtTime(30, audioCtx.currentTime);
            compressorNode.ratio.setValueAtTime(6, audioCtx.currentTime);
            compressorNode.attack.setValueAtTime(0.00, audioCtx.currentTime);
            compressorNode.release.setValueAtTime(0.1, audioCtx.currentTime);
            
            gainNode.gain.setValueAtTime(currentGainValue, audioCtx.currentTime);
            if (slider) {
                slider.disabled = false;
                slider.style.opacity = '1.0';
            }
        } else {
            // 컴프레서 끔: 압축 해제 및 게인 1.0배 복구
            compressorNode.threshold.setValueAtTime(0, audioCtx.currentTime);
            gainNode.gain.setValueAtTime(1.0, audioCtx.currentTime);
            if (slider) {
                slider.disabled = true;
                slider.style.opacity = '0.3';
            }
        }

        const icon = btn.querySelector('#compressor-icon');
        icon.style.stroke = isCompressorOn ? '#00ffaa' : '#ffffff';
        icon.style.opacity = isCompressorOn ? '1.0' : '0.5';
    }

    // --- UI 로직 (치지직 나이프 및 전용 슬라이더 구조 주입) ---
    function injectCompressorButton() {
        if (document.getElementById('chzzk-audio-compressor-btn')) return;

        const target = document.querySelector('.pzp-pc__bottom-buttons-left');
        if (!target) return;

        // 버튼과 슬라이더를 감싸는 전체 컨테이너 생성
        const wrapper = document.createElement('div');
        wrapper.className = 'pzp-pc__volume-control knife-comp';
        wrapper.style.display = 'flex';
        wrapper.style.alignItems = 'center';
        wrapper.style.gap = '8px';

        // 1. 오디오 컴프레서 토글 버튼
        const btn = document.createElement('button');
        btn.id = 'chzzk-audio-compressor-btn';
        btn.className = 'pzp-button pzp-pc__button';
        btn.style.cssText = "display: inline-flex !important; align-items: center !important; justify-content: center !important; height: 34px !important; width: 34px !important; background: transparent !important; border: none !important; cursor: pointer !important;";

        btn.innerHTML = `
            <span class="pzp-ui-icon">
                <svg id="compressor-icon" style="opacity: 0.5; transition: 0.2s;" width="22" height="22" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
                    <path d="M3 10v4M6 6v12M11 3v18M16 8v8M21 11v2"></path>
                </svg>
            </span>
            <span class="pzp-button__tooltip pzp-button__tooltip--top">오디오 컴프레서</span>
        `;
        btn.onclick = () => toggleCompressor(btn);
        wrapper.appendChild(btn);

        // 2. 치지직 UI 스타일을 차용한 게인 조절 슬라이더 영역
        const sliderContainer = document.createElement('div');
        sliderContainer.className = 'pzp-pc__volume-slider pzp-ui-slider--volume knife-gain-slider';
        sliderContainer.style.cssText = "display: flex; align-items: center; width: 80px; height: 34px; position: relative;";

        const slider = document.createElement('input');
        slider.id = 'chzzk-compressor-gain-slider';
        slider.type = 'range';
        slider.min = '1.0';
        slider.max = '3.0';
        slider.step = '0.1';
        slider.value = currentGainValue.toString();
        slider.disabled = !isCompressorOn; // 초기 상태에 맞춰 활성화 여부 결정
        slider.style.cssText = `
            width: 100%;
            height: 6px;
            cursor: pointer;
            accent-color: #00ffaa;
            opacity: ${isCompressorOn ? '1.0' : '0.3'};
            transition: opacity 0.2s;
        `;

        // 슬라이더 툴팁 구현
        const tooltip = document.createElement('span');
        tooltip.className = 'pzp-button__tooltip pzp-button__tooltip--top';
        tooltip.innerText = `게인 보정: ${currentGainValue}x`;
        sliderContainer.appendChild(tooltip);

        // 슬라이더 이벤트 바인딩 (드래그 시 실시간 게인 노드 값 변경)
        slider.oninput = (e) => {
            currentGainValue = parseFloat(e.target.value);
            tooltip.innerText = `게인 보정: ${currentGainValue.toFixed(1)}x`;

            if (isCompressorOn && gainNode) {
                gainNode.gain.setValueAtTime(currentGainValue, audioCtx.currentTime);
            }
        };

        sliderContainer.appendChild(slider);
        wrapper.appendChild(sliderContainer);

        // 치지직 오리지널 볼륨 컨트롤러 바로 뒤에 배치
        const vol = target.querySelector('.pzp-pc__volume-control');
        if (vol) {
            vol.after(wrapper);
        } else {
            target.appendChild(wrapper);
        }
    }

    // --- 미리보기 로직 ---
    const previewContainer = document.createElement('div');
    Object.assign(previewContainer.style, { position: 'fixed', zIndex: '99999', display: 'none', width: '280px', height: '157px', backgroundColor: '#000', borderRadius: '6px', overflow: 'hidden', boxShadow: '0 6px 20px rgba(0,0,0,0.5)', border: '2px solid #00ffaa', pointerEvents: 'none' });
    const previewVideo = document.createElement('video');
    previewVideo.muted = true; previewVideo.autoplay = true; previewVideo.style.cssText = "width: 100%; height: 100%; object-fit: cover;";
    previewContainer.appendChild(previewVideo);
    document.body.appendChild(previewContainer);

    async function showPreview(sidebarItem, channelId, previewContainer, previewVideo) {
        if (channelId === currentChannel) return;
        currentChannel = channelId;

        abortController?.abort();
        abortController = new AbortController();

        if (!liveCache.has(channelId)) {
            try {
                const response = await fetch(`https://api.chzzk.naver.com/service/v2/channels/${channelId}/live-detail`, { signal: abortController.signal });
                const data = await response.json();
                if (data?.content?.livePlaybackJson) {
                    const playback = typeof data.content.livePlaybackJson === 'string' ? JSON.parse(data.content.livePlaybackJson) : data.content.livePlaybackJson;
                    if (playback?.media?.[0]?.path) liveCache.set(channelId, playback.media[0].path);
                }
            } catch (e) {
                if (e.name !== 'AbortError') console.error(e);
                return;
            }
        }

        const streamUrl = liveCache.get(channelId);
        if (!streamUrl || currentPreviewUrl === streamUrl) return;
        currentPreviewUrl = streamUrl;

        if (!hls) {
            hls = new Hls({ maxBufferLength: 1, enableWorker: true, lowLatencyMode: true });
            hls.attachMedia(previewVideo);
        }
        hls.loadSource(streamUrl);

        const rect = sidebarItem.getBoundingClientRect();
        previewContainer.style.top = `${Math.max(10, Math.min(rect.top, window.innerHeight - 170))}px`;
        previewContainer.style.left = `${rect.right + 15}px`;
        previewContainer.style.display = 'block';
    }

    // 캡처 단계에서 이벤트 감지 및 처리
    document.addEventListener('pointerenter', (e) => {
        if (!e.target || typeof e.target.closest !== 'function') return;

        const linkElem = e.target.closest('a[href*="/live/"]');
        const sidebarItem = linkElem?.closest('li');
        if (!sidebarItem) return;

        const channelId = linkElem.href.split('/live/')[1]?.split('?')[0];
        if (hoverTimeout) clearTimeout(hoverTimeout);
        hoverTimeout = setTimeout(() => showPreview(sidebarItem, channelId, previewContainer, previewVideo), 250);
    }, true);

    document.addEventListener('pointerleave', (e) => {
        if (!e.target || typeof e.target.closest !== 'function') return;

        if (e.target.closest('a[href*="/live/"]')) {
            clearTimeout(hoverTimeout);
            previewContainer.style.display = 'none';
            currentChannel = null;
            currentPreviewUrl = null;
            abortController?.abort();
        }
    }, true);

    // 주기적 실행부 (버튼 및 슬라이더 주입 감시)
    setInterval(injectCompressorButton, 1000);

    // 30초마다 자동 새로고침 버튼 클릭
    setInterval(() => {
        const refreshBtn = document.querySelector('button[aria-label="새로고침"]');
        if (refreshBtn) {
            refreshBtn.click();
            console.log('[치지직 툴] 30초 주기 새로고침 버튼을 클릭했습니다.');
        }
    }, 30000);

})();
